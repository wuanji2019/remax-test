import { createSlice } from '@reduxjs/toolkit';
import type { PayloadAction } from '@reduxjs/toolkit';

export interface CounterState {
  statusLoading: boolean;
  statusToast: boolean;
  statusToastMessage: string;
}

const initialState: CounterState = {
  statusLoading: false,
  statusToast: false,
  statusToastMessage: ''
};

export const STATUS_FEATURE_KEY = 'state';

const { reducer: statusReducer, actions } = createSlice({
   name: STATUS_FEATURE_KEY,
  initialState,
  // The `reducers` field lets us define reducers and generate associated actions
  reducers: {
    setLoading(state, action: PayloadAction<boolean>) {
      state.statusLoading = action.payload;
    },
    setToast(state, action: PayloadAction<any>) {
      const { message, active } = action.payload;
      state.statusToast = active;
      state.statusToastMessage = message;
    }
  }
  // The `extraReducers` field lets the slice handle actions defined elsewhere,
  // including actions generated by createAsyncThunk or in other slices.
  extraReducers: (builder) => {
    builder
      .addCase(incrementAsync.pending, (state) => {
        state.status = 'loading';
      })
      .addCase(incrementAsync.fulfilled, (state, action) => {
        state.status = 'idle';
        state.value += action.payload;
      });
  }
});

export const { setLoading, setToast } = actions;

// The function below is called a selector and allows us to select a value from
// the state. Selectors can also be defined inline where they're used instead of
// in the slice file. For example: `useSelector((state: RootState) => state.counter.value)`
export const statusToast = (state) => state.counter.statusToast;
